  return (

      shadow="none"

      {/* Clock Content */}
      <CardBody className="flex flex-col items-center justify-center p-8 ">
        <View className="flex flex-col items-center justify-center border-[1rem] rounded-4xl overflow-hidden">
          <Calendar
            value={value}
            color="danger"
            showShadow={false}
            onChange={setValue}
            // className="light border-none shadow-none"
          />
        </View>
      </CardBody>
    </Card>
  )
}
